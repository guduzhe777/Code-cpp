#### //产生全排列数字 

//并不是一个一个循环产生 太麻烦

//可以在取值范围内放进同一个数组内 然后判断数组每一个元素都不同

//（基于同一个数组 用 每个元素不同 利用双指针利于计算）大大节省了时间复杂度 

~~~
//放进数组
while (x / 10)
        {
            arr[j] = x % 10;
            x /= 10;
            j++;
        }
~~~

~~~
//双指针判断数组中的每个元素都不同
int dis(long long arr[], int len)
{
    for (int i = 0; i < len; i++)
    {
        for (int j = i + 1; j < len; j++)
        {
            if (arr[i] == arr[j]) //凡是有相等
            {
                return 0;
            }
        }
    }
    return 1;
}
~~~
