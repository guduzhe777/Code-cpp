~~~
for(int i = 0; i<s.length(); i++)
    {
        orig[i] = s[i] - '0';
        change[i] = orig[i]*2;//change[i] 原数字的double
    }
    for(size_t i = s.length()-1; i>0; i--)
    {
        while(change[i]>9)
        {
            change[i]-=10;
            change[i-1] += 1;
        }
    }//得到进位数字
~~~
